#
# This file contains the definition of the JSON payload format for the temperature sensing data.
#

# JSON data structures may not conatain comments, so the comments in this file are for documentation purposes only
# and should not be included in the actual JSON payload.

# Payload version history:
# Version 1.0, 2026-02-26:  Initial proposal for JSON payload format
# Version 1.1, 2026-02-28:  Added proposal for timestamp field when storing datasets on disk
# Version 1.2, 2026-03-01:  Removed payload version as part of the payload (see backlog on how payload versioning could be handled in the future)
# Version 1.3, 2026-03-08:  The payload shall contain only configured sensors (see details about unconfigured sensors below)

# JSON Payload Formatting Proposal:
{

	"client": "tmc0",		# Client name
	"sb_nr": 0,			    # Sensor bank number
	"ds_nr": 0,			    # Data set number, starts with 0 for the first set published and gets incremented with each set
	"ts_dat":			    # Temperature sensing data are following with a maximum of 8 name/value pairs, e.g. s0 to s7 for 8 sensors
	{
		"s0": 20.00,		# Name of the first configured sensor in the currrent sensor bank of the client,
		"s1": 21.10,		# sensornames can be max. 8 characters long, no space character may be part of the name
		"s2": 22.20,		# Temperature values are given as signed integers (not enclosed in "" -> no character strings are allowed)
		"s3": 23.30,		# If an unconfigured sensor is encountered during payload assembly, that name/value pair will get skipped,
		"s4": 24.40,        # e.g. if only 4 sensors are configured, the payload will only contain s0 to s3		 
		"s5": 25.50,		# If a configured sensor does not deliver data, the value is supposed to read "99.99"  
		"s6": 26.60,
		"s7": 27.70
	}
}

# When datasets are going to be stored on disk, a timestamp shall get added.
# It shall get inserted between ds_nr and ts_dat, e.g.
# "time": "2026-01-28 14:36:12" -> is there any other format conceivable besides string?

# Compactformat:
{"client":"tmc0","sb_nr":0,"ds_nr":0,"ts_dat":{"s0":20,"s1":21.1,"s2":22.2,"s3":23.3,"s4":24.4,"s5":25.5,"s6":26.6,"s7":27.7}}
